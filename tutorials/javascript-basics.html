JavaScript Basics

JavaScript is a versatile programming language that enables interactive and dynamic web applications.

Language Basics

Core concepts and syntax of JavaScript:


// Variables and Data Types
let name = "John";              // String
const age = 25;                 // Number
let isStudent = true;           // Boolean
let scores = [95, 88, 76];      // Array
let person = {                  // Object
    name: "John",
    age: 25
};

// Template Literals
let greeting = `Hello ${name}, you are ${age} years old`;

// Conditional Statements
if (age >= 18) {
    console.log("Adult");
} else {
    console.log("Minor");
}

// Loops
for (let i = 0; i < scores.length; i++) {
    console.log(scores[i]);
}

// Modern For...of Loop
for (let score of scores) {
    console.log(score);
}

// Switch Statement
let role = "admin";
switch (role) {
    case "admin":
        console.log("Full access");
        break;
    case "user":
        console.log("Limited access");
        break;
    default:
        console.log("No access");
}

Functions & Scope

Different ways to define and use functions:


// Function Declaration
function greet(name) {
    return `Hello, ${name}!`;
}

// Function Expression
const add = function(a, b) {
    return a + b;
};

// Arrow Function
const multiply = (a, b) => a * b;

// Default Parameters
function createUser(name, role = "user") {
    return { name, role };
}

// Rest Parameters
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

// Closure Example
function counter() {
    let count = 0;
    return {
        increment: () => ++count,
        decrement: () => --count,
        getCount: () => count
    };
}

// IIFE (Immediately Invoked Function Expression)
(function() {
    console.log("This runs immediately!");
})();

Objects & Classes

Object-oriented programming in JavaScript:


// Object Creation
const person = {
    name: "John",
    age: 25,
    greet() {
        return `Hello, I'm ${this.name}`;
    }
};

// Class Declaration
class User {
    constructor(name, email) {
        this.name = name;
        this.email = email;
    }

    // Instance Method
    getInfo() {
        return `${this.name} (${this.email})`;
    }

    // Static Method
    static isValidEmail(email) {
        return email.includes('@');
    }
}

// Inheritance
class Admin extends User {
    constructor(name, email, role) {
        super(name, email);
        this.role = role;
    }

    hasPermission(task) {
        return this.role === 'admin';
    }
}

// Getters and Setters
class Circle {
    constructor(radius) {
        this._radius = radius;
    }

    get area() {
        return Math.PI * this._radius ** 2;
    }

    set radius(value) {
        if (value > 0) {
            this._radius = value;
        }
    }
}

Arrays & Methods

Working with arrays and modern array methods:


// Array Creation and Basic Methods
let fruits = ["apple", "banana", "orange"];
fruits.push("grape");           // Add to end
fruits.pop();                   // Remove from end
fruits.unshift("lemon");        // Add to start
fruits.shift();                 // Remove from start

// Modern Array Methods
let numbers = [1, 2, 3, 4, 5];

// map - Transform elements
let doubled = numbers.map(n => n * 2);

// filter - Select elements
let evens = numbers.filter(n => n % 2 === 0);

// reduce - Accumulate values
let sum = numbers.reduce((total, n) => total + n, 0);

// find - Get first matching element
let found = numbers.find(n => n > 3);

// some/every - Check conditions
let hasEven = numbers.some(n => n % 2 === 0);
let allPositive = numbers.every(n => n > 0);

// Spread Operator
let combined = [...fruits, ...numbers];

// Destructuring
let [first, second, ...rest] = numbers;

Asynchronous Programming

Handling asynchronous operations:


// Promises
function fetchData() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const data = { id: 1, name: "John" };
            resolve(data);
            // reject(new Error("Failed to fetch"));
        }, 1000);
    });
}

// Using Promises
fetchData()
    .then(data => console.log(data))
    .catch(error => console.error(error));

// Async/Await
async function getData() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
    }
}

// Promise.all
async function getAllData() {
    const [users, posts] = await Promise.all([
        fetch('/api/users').then(r => r.json()),
        fetch('/api/posts').then(r => r.json())
    ]);
    return { users, posts };
}

DOM Manipulation

Working with the Document Object Model:


// Selecting Elements
const element = document.getElementById('myId');
const elements = document.getElementsByClassName('myClass');
const query = document.querySelector('.myClass');
const queryAll = document.querySelectorAll('.item');

// Creating and Adding Elements
const div = document.createElement('div');
div.className = 'new-element';
div.textContent = 'New Content';
document.body.appendChild(div);

// Modifying Elements
element.innerHTML = 'New content';
element.setAttribute('data-id', '123');
element.style.color = 'blue';
element.classList.add('active');
element.classList.remove('inactive');
element.classList.toggle('visible');

// Traversing the DOM
const parent = element.parentNode;
const children = element.children;
const next = element.nextElementSibling;
const prev = element.previousElementSibling;

// Fragment for Better Performance
const fragment = document.createDocumentFragment();
for (let i = 0; i < 3; i++) {
    const item = document.createElement('div');
    item.textContent = `Item ${i}`;
    fragment.appendChild(item);
}
document.body.appendChild(fragment);

Event Handling

Managing browser and user events:


// Event Listeners
element.addEventListener('click', function(event) {
    console.log('Clicked!', event);
});

// Event Object Properties
element.addEventListener('click', (e) => {
    e.preventDefault();     // Prevent default behavior
    e.stopPropagation();   // Stop event bubbling
    console.log(e.target); // Element that triggered event
});

// Common Events
document.addEventListener('DOMContentLoaded', () => {
    console.log('DOM fully loaded');
});

window.addEventListener('load', () => {
    console.log('Page fully loaded');
});

// Form Events
form.addEventListener('submit', (e) => {
    e.preventDefault();
    const formData = new FormData(form);
    console.log(Object.fromEntries(formData));
});

// Custom Events
const customEvent = new CustomEvent('myEvent', {
    detail: { message: 'Hello!' }
});
element.dispatchEvent(customEvent);

// Event Delegation
document.addEventListener('click', (e) => {
    if (e.target.matches('.btn')) {
        console.log('Button clicked');
    }
});

Error Handling

Handling and managing errors:


// Try-Catch Block
try {
    // Code that might throw an error
    throw new Error('Something went wrong');
} catch (error) {
    console.error('Error:', error.message);
} finally {
    console.log('This always runs');
}

// Custom Error Types
class ValidationError extends Error {
    constructor(message) {
        super(message);
        this.name = 'ValidationError';
    }
}

// Async Error Handling
async function fetchUser(id) {
    try {
        const response = await fetch(`/api/users/${id}`);
        if (!response.ok) {
            throw new Error('User not found');
        }
        return await response.json();
    } catch (error) {
        console.error('Failed to fetch user:', error);
        throw error; // Re-throw to handle it upstream
    }
}

// Global Error Handling
window.onerror = function(msg, url, line, col, error) {
    console.error('Global error:', { msg, url, line, col, error });
    return false; // Let default error handling continue
};

// Promise Error Handling
Promise.reject(new Error('Promise failed'))
    .catch(error => {
        console.error('Caught:', error);
    });

Best Practices

  • Use strict mode ('use strict') to catch common mistakes
  • Always declare variables with const or let, avoid var
  • Use meaningful variable and function names
  • Keep functions small and focused
  • Comment complex logic but write self-documenting code
  • Use modern ES6+ features when possible
  • Handle errors appropriately
  • Use consistent code formatting
  • Optimize DOM manipulations
  • Use proper event delegation
  • Cache DOM queries for better performance
  • Follow the DRY (Don't Repeat Yourself) principle
